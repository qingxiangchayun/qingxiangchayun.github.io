$(function(){
    var $htmlArea = $('#J_html_area');
    var $jsArea = $('#J_js_area');

    $('#J_tojs_btn').on('click',function(){
        var htmlStr = $htmlArea.val();
    
        var jsStr =  htmlStr.split('\n')                            // 以\n split 
                            .join("\',\'")                          // 以 ',' join  会在每行标签末尾加上 ', 在每行标签首部加上 '
                            .replace(/>\',\'/g,">\',\n\'")          // 将 >',' 替换为 >',n' 相当于将最开始的 \n 在加回来 正则中加上 > 可以避免替换范围过大 将html 文本中',' 替换错误
                            .replace(/(\')(\t+|\s+)/g,'$2$1');      // 将 '(\t)(\s) 替换为 (\t)(\s)'

        jsStr = "[\'" + jsStr + "\'].join(\'\')";

        $jsArea.val(jsStr);
    });
});
