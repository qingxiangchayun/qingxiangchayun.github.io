Tool -- HTML TO Array

$(function(){
	var $htmlArea = $('#J_html_area');
	var $arrayArea = $('#J_array_area');

	$('#J_toArray_btn').on('click',function(){
		var htmlStr = $htmlArea.val();

	
		var arrStr = htmlStr.split('\n')
							.join("\',\'")
							.replace(/>\',\'/g,">\',\n\'")
							.replace(/(\')(\t+|\s+)/g,'$2$1');

		arrStr = "[\'" + arrStr + "\'].join(\'\')";

		$arrayArea.val(arrStr);
	});

});
